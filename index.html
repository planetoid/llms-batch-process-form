  1. Install a CORS proxy browser extension (temporary solution)
  2. Or use our Node.js proxy server (recommended):
// server.js
const express = require('express');
const cors = require('cors');
const { createProxyMiddleware } = require('http-proxy-middleware');

const app = express();
app.use(cors());

app.use('/api', createProxyMiddleware({
    target: 'https://api.anthropic.com',
    changeOrigin: true,
    pathRewrite: {
        '^/api': '/v1'
    }
}));

app.listen(3000);
